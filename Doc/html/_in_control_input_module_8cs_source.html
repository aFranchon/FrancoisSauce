810  }
811 
812  // search for the control that will receive the press
813  // if we can't find a press handler set the press

816 
817  // didnt find a press handler... search for a click handler
818  if (newPressed == null)
819  newPressed = ExecuteEvents.GetEventHandler<IPointerClickHandler>( currentOverGo );

822 
823  if (newPressed == pointerEvent.lastPress)

827  ++pointerEvent.clickCount;
828  else
829  pointerEvent.clickCount = 1;
830

832  }
833  else

835  pointerEvent.clickCount = 1;
836  }

839  pointerEvent.rawPointerPress = currentOverGo;
840

844  pointerEvent.pointerDrag = ExecuteEvents.GetEventHandler<IDragHandler>( currentOverGo );
