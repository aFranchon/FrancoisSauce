466  public void Commit( ulong updateTick, float deltaTime )
467  {
468  // We need to do some processing for known controllers to ensure all
469  // the various control aliases holding directional values are calculated
470  // optimally with circular deadzones and then set properly everywhere.
471  if (IsKnown)
472  {
473  ProcessLeftStick( updateTick, deltaTime );
474  ProcessRightStick( updateTick, deltaTime );
475  ProcessDPad( updateTick, deltaTime );
476  }
477 
478  // Commit all control values.
479  var controlsCount = Controls.Count;
480  for (var i = 0; i < controlsCount; i++)
481  {
482  var control = Controls[i];
483  if (control != null)
484  {
485  control.Commit();
486  }
487  }
488 
489  // Calculate the Command alias state for known devices and commit it.
490  if (IsKnown)
491  {
492  var passive = true;
493  var pressed = false;
494  for (var i = (int) InputControlType.Back; i <= (int) InputControlType.Minus; i++)
495  {
496  var control = ControlsByTarget[i];
497  if (control != null && control.IsPressed)
498  {
499  pressed = true;
500  if (!control.Passive)
501  {
502  passive = false;
503  }
504  }
505  }
506 
507  Command.Passive = passive;
508  Command.CommitWithState( pressed, updateTick, deltaTime );
509  }
510 
511  // If any non-passive controls provide input, flag the device active.
512  IsActive = false;
513  for (var i = 0; i < controlsCount; i++)
514  {
515  var control = Controls[i];
516  if (control != null && control.HasInput && !control.Passive)
517  {
518  LastInputTick = updateTick;
519  IsActive = true;
520  }
521  }
522  }
