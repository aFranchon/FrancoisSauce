52 
53  foreach (var file in aotDir.GetFiles("*.dll"))
54  {

56  if (isPackage)
57  {
58  path = SirenixAssetPaths.SirenixAssembliesPath.TrimEnd('\\', '/') + "/" + path.Substring(assemblyDir.Length);
59  }
60  else
61  {
62  path = path.Substring(projectAssetsPath.Length + 1);
63  }
64 
65  aotAssemblies.Add(path);

68  foreach (var file in jitDir.GetFiles("*.dll"))
