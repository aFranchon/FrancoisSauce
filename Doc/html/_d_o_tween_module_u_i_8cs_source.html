97  public static Sequence DOGradientColor(this Image target, Gradient gradient, float duration)
98  {
99  Sequence s = DOTween.Sequence();
100  GradientColorKey[] colors = gradient.colorKeys;
101  int len = colors.Length;
102  for (int i = 0; i < len; ++i) {
103  GradientColorKey c = colors[i];
104  if (i == 0 && c.time <= 0) {
105  target.color = c.color;
106  continue;
107  }
108  float colorDuration = i == len - 1
109  ? duration - s.Duration(false) // Verifies that total duration is correct
110  : duration * (i == 0 ? c.time : c.time - colors[i - 1].time);
111  s.Append(target.DOColor(c.color, colorDuration).SetEase(Ease.Linear));
112  }
113  s.SetTarget(target);
114  return s;
115  }

389  public static Sequence DOJumpAnchorPos(this RectTransform target, Vector2 endValue, float jumpPower, int numJumps, float duration, bool snapping = false)
390  {
391  if (numJumps < 1) numJumps = 1;
392  float startPosY = 0;
393  float offsetY = -1;
394  bool offsetYSet = false;
395 
396  // Separate Y Tween so we can elaborate elapsedPercentage on that insted of on the Sequence
397  // (in case users add a delay or other elements to the Sequence)
398  Sequence s = DOTween.Sequence();
399  Tween yTween = DOTween.To(() => target.anchoredPosition, x => target.anchoredPosition = x, new Vector2(0, jumpPower), duration / (numJumps * 2))
400  .SetOptions(AxisConstraint.Y, snapping).SetEase(Ease.OutQuad).SetRelative()
401  .SetLoops(numJumps * 2, LoopType.Yoyo)
402  .OnStart(()=> startPosY = target.anchoredPosition.y);
403  s.Append(DOTween.To(() => target.anchoredPosition, x => target.anchoredPosition = x, new Vector2(endValue.x, 0), duration)
404  .SetOptions(AxisConstraint.X, snapping).SetEase(Ease.Linear)
405  ).Join(yTween)
406  .SetTarget(target).SetEase(DOTween.defaultEaseType);
407  s.OnUpdate(() => {
408  if (!offsetYSet) {
409  offsetYSet = true;
410  offsetY = s.isRelative ? endValue.y : endValue.y - startPosY;
411  }
412  Vector2 pos = target.anchoredPosition;
413  pos.y += DOVirtual.EasedValue(0, offsetY, s.ElapsedDirectionalPercentage(), Ease.OutQuad);
414  target.anchoredPosition = pos;
415  });
416  return s;
417  }

621  public static Vector2 SwitchToRectTransform(RectTransform from, RectTransform to)
622  {
623  Vector2 localPoint;
624  Vector2 fromPivotDerivedOffset = new Vector2(from.rect.width * 0.5f + from.rect.xMin, from.rect.height * 0.5f + from.rect.yMin);
625  Vector2 screenP = RectTransformUtility.WorldToScreenPoint(null, from.position);
626  screenP += fromPivotDerivedOffset;
627  RectTransformUtility.ScreenPointToLocalPointInRectangle(to, screenP, null, out localPoint);
628  Vector2 pivotDerivedOffset = new Vector2(to.rect.width * 0.5f + to.rect.xMin, to.rect.height * 0.5f + to.rect.yMin);
629  return to.anchoredPosition + localPoint - pivotDerivedOffset;
630  }
631  }
